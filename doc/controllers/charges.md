        recurrency_cycle='"first" or "subsequent"'
    ),
    cash=CreateCashPaymentRequest(
        description='description0',
        confirm=False
    ),
    bank_transfer=CreateBankTransferPaymentRequest(
        bank='bank0',
        retries=236
    ),
    private_label=CreatePrivateLabelPaymentRequest(
        installments=1,

    charge_id,
